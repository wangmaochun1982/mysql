#### 2、增量脚本

```bash
[root@mydb backup]# cat incrbackup.sh
#!/bin/bash

BAK_DIR_ROOT="/data/backup/xtrabackup"
MYSQL_USERNAME="root"
MYSQL_PASSWORD="Super@99"
MYSQL_CNF="/etc/my.cnf"

BAK_FULL_DIR=$BAK_DIR_ROOT/mysqlbackup`(date +%Y%m%d)`/mysqlfullbackup
BAK_LOG="$BAK_DIR_ROOT/mysqlbackup`(date +%Y%m%d)`/backup.log"

ti=`(date +%H)`
ti1=$((10#$ti - 0))
ti2=$((ti1 - 1))

#BASE_DIR=$BAK_DIR_ROOT/mysqlbackup`(date +%Y%m%d)`/mysqlincbackup${ti2}
INC_DIR=$BAK_DIR_ROOT/mysqlbackup`(date +%Y%m%d)`/mysqlincbackup${ti1}


if [[ ! -d ${INC_DIR} ]]; then
    mkdir -p ${INC_DIR}
fi

if [[ ${ti2} -eq 0 ]];then
    BASE_DIR=${BAK_FULL_DIR}
else
    BASE_DIR=$BAK_DIR_ROOT/mysqlbackup`(date +%Y%m%d)`/mysqlincbackup${ti2}
fi

xtrabackup --defaults-file=${MYSQL_CNF} \
           --user=${MYSQL_USERNAME} \
           --password=${MYSQL_PASSWORD} \
           --backup --stream=xbstream \
           --target-dir=${INC_DIR} \
           --incremental-basedir="${BASE_DIR}/LSN_INFO" \
           --extra-lsndir="${INC_DIR}/LSN_INFO" > "${INC_DIR}/mysqlincbackup${ti1}.xbstream" 2>> ${BAK_LOG}
[root@mydb backup]#


```

1. ❌ 不再用“时间推算上一个备份”

4. ❌ 不再依赖 `LSN_INFO`
5. ✅ 自动跨天、跨小时都不影响

# 一、全备脚本（优化版）

👉 基本不用大改，只做小优化

```bash
#!/bin/bash

BAK_DIR_ROOT="/data/backup/xtrabackup"
MYSQL_USERNAME="root"
MYSQL_PASSWORD="Super@99"
MYSQL_CNF="/etc/my.cnf"
MYSQL_SOCKET="/data/mysql/mysql.sock"

TODAY=$(date +%Y%m%d)

BAK_BASE_DIR=${BAK_DIR_ROOT}/${TODAY}
BAK_FULL_DIR=${BAK_BASE_DIR}/full
BAK_LOG="${BAK_BASE_DIR}/backup.log"

mkdir -p ${BAK_FULL_DIR}

xtrabackup --defaults-file=${MYSQL_CNF} \
           --user=${MYSQL_USERNAME} \
           --password=${MYSQL_PASSWORD} \
           --socket=${MYSQL_SOCKET} \
           --backup \
           --compress=zstd \
           --compress-threads=2 \
           --target-dir=${BAK_FULL_DIR} \
           --throttle=100 \
           --parallel=2 \
           2>> ${BAK_LOG}

# ✔ 失败检测（建议一定加）
if [ $? -ne 0 ]; then
    echo "FULL BACKUP FAILED at $(date)" >> ${BAK_LOG}
    exit 1
fi
```

# 二、增量脚本（核心改造版 ✅）

👉 重点在这里（保证不断链）

```bash
#!/bin/bash

BAK_DIR_ROOT="/data/backup/xtrabackup"
MYSQL_USERNAME="root"
MYSQL_PASSWORD="Super@99"
MYSQL_CNF="/etc/my.cnf"

TODAY=$(date +%Y%m%d)
HOUR=$(date +%H)

BAK_BASE_DIR=${BAK_DIR_ROOT}/${TODAY}
INC_DIR=${BAK_BASE_DIR}/inc_${HOUR}
BAK_LOG="${BAK_BASE_DIR}/backup.log"

mkdir -p ${INC_DIR}

# ✅ 找最近一次备份（关键点）
BASE_DIR=$(ls -dt ${BAK_DIR_ROOT}/*/* 2>/dev/null | grep -E '/(full|inc_)' | head -1)

# ❗ 如果没找到（第一次跑），就用当天全备
if [[ -z "${BASE_DIR}" ]]; then
    BASE_DIR=${BAK_BASE_DIR}/full
fi

echo "BASE_DIR=${BASE_DIR}" >> ${BAK_LOG}

xtrabackup --defaults-file=${MYSQL_CNF} \
           --user=${MYSQL_USERNAME} \
           --password=${MYSQL_PASSWORD} \
           --backup \
           --parallel=2 \
           --throttle=100 \
           --compress=zstd \
           --compress-threads=2 \
           --target-dir=${INC_DIR} \
           --incremental-basedir=${BASE_DIR} \
           2>> ${BAK_LOG}

# ✔ 失败检测
if [ $? -ne 0 ]; then
    echo "INCREMENTAL BACKUP FAILED at $(date)" >> ${BAK_LOG}
    exit 1
fi
```

# ✅ 一、先全部解压（必须）

## 🟢 1. 解压 full

```
xtrabackup --decompress \
  --parallel=2 \
  --target-dir=/data/backup/xtrabackup/20260415/full
```

------

## 🟢 2. 解压每个增量（都要）

```
xtrabackup --decompress \
  --parallel=2 \
  --target-dir=/data/backup/xtrabackup/20260415/inc_01

xtrabackup --decompress \
  --parallel=2 \
  --target-dir=/data/backup/xtrabackup/20260415/inc_02

xtrabackup --decompress \
  --parallel=2 \
  --target-dir=/data/backup/xtrabackup/20260415/inc_03
```

------

## 🟡（可选）删除压缩文件，节省空间

```
find /data/backup/xtrabackup/20260415/ -name "*.zst" -delete
```

------

# ✅ 二、prepare（核心步骤）

------

## 🟢 3. prepare full（只做 redo，不回滚）

```
xtrabackup --prepare \
  --apply-log-only \
  --target-dir=/data/backup/xtrabackup/20260415/full
```

------

## 🟢 4. 合并 inc_01

```
xtrabackup --prepare \
  --apply-log-only \
  --target-dir=/data/backup/xtrabackup/20260415/full \
  --incremental-dir=/data/backup/xtrabackup/20260415/inc_01
```

------

## 🟢 5. 合并 inc_02

```
xtrabackup --prepare \
  --apply-log-only \
  --target-dir=/data/backup/xtrabackup/20260415/full \
  --incremental-dir=/data/backup/xtrabackup/20260415/inc_02
```

------

## 🔴 6. 合并 inc_03（最后一个！不要加 apply-log-only）

```
xtrabackup --prepare \
  --target-dir=/data/backup/xtrabackup/20260415/full \
  --incremental-dir=/data/backup/xtrabackup/20260415/inc_03
```

------

# ❗关键点总结（必须记住）

- 前面全部：`--apply-log-only`
- **最后一个增量：绝对不能加**

# ✅ 三、恢复回 MySQL 数据目录


# ⚠️ 最后提醒你3个最容易翻车的点

1. ❌ 忘了解压 → prepare直接报错
2. ❌ 最后一步还加 `--apply-log-only` → 数据不一致
3. ❌ 增量顺序错 → 恢复失败

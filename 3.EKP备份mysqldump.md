```bash
#!/bin/bash
# MySQL 流式不落地 + 不生成临时 SQL 文件
source /etc/profile
source ~/.bash_profile
set -o nounset

# ====================== 配置项（请修改） ======================
user="root"
password="xxxxxxx"
host="localhost"
port="3306"
socket="/data/mysql/mysql.sock"
# 要备份的数据库
db_list=("ekp" "kk" "aip")
# 备份保留天数
keep_days=15
# 备份路径
backup_path="/backup/mysql_bak"
backup_log="${backup_path}/backup.log"
# =============================================================

# 初始化目录
init_dir() {
    [ ! -d "$backup_path" ] && mkdir -p "$backup_path"
    # 检查磁盘空间
    free_space=$(df -BG "$backup_path" | awk 'NR==2 {print $4}' | tr -d 'G')
    if [ "$free_space" -lt 50 ]; then
        echo "[$(date '+%Y-%m-%d %H:%M:%S')] 错误：备份目录空间不足" >> "$backup_log"
        exit 1
    fi
}

# 清理历史备份
clean_old_backup() {
    find "$backup_path" -type f -name "*.sql.gz" -mtime +$keep_days -delete > /dev/null 2>&1
    echo "[$(date '+%Y-%m-%d %H:%M:%S')] 清理${keep_days}天前旧备份完成" >> "$backup_log"
}

# gzip 流式备份核心（不生成临时SQL，直接压缩，速度最快）
mysql_backup() {
    local db_name=$1
    local backup_time=$(date +%Y-%m-%d_%H-%M-%S)
    local backup_file="${backup_path}/${db_name}_${backup_time}.sql.gz"
    
    echo "[$(date '+%Y-%m-%d %H:%M:%S')] 开始备份：$db_name" >> "$backup_log"

    # 核心优化：mysqldump 直接输出 → gzip 压缩 → 生成.gz文件（无临时文件）
    mysqldump -S "$socket" -P "$port" -u"$user" -p"$password" \
        --single-transaction \
        --default-character-set=utf8mb4 \
        --opt \
        --skip-lock-tables \
        --set-gtid-purged=OFF \
        --events --triggers --routines \
        --databases "$db_name" \
    | gzip > "$backup_file"

    # 判断结果
    if [ $? -eq 0 ]; then
        local file_size=$(du -sh "$backup_file" | awk '{print $1}')
        echo "[$(date '+%Y-%m-%d %H:%M:%S')] 备份成功：$db_name | 大小：$file_size" >> "$backup_log"
    else
        rm -f "$backup_file"
        echo "[$(date '+%Y-%m-%d %H:%M:%S')] 备份失败：$db_name" >> "$backup_log"
    fi
}

# ====================== 执行 ======================
init_dir
clean_old_backup

echo -e "\n==================== 备份开始：$(date '+%Y-%m-%d %H:%M:%S') ====================" >> "$backup_log"

for db in "${db_list[@]}"; do
    mysql_backup "$db"
done

echo "==================== 备份结束：$(date '+%Y-%m-%d %H:%M:%S') ====================\n" >> "$backup_log"
```

```bash
#!/bin/bash
#删除15天以前备份
source /etc/profile        #加载系统环境变量
source ~/.bash_profile    #加载用户环境变量
set -o nounset             #引用未初始化变量时退出
#set -o errexit            #执行shell命令遇到错误时退出

user="root"
password="SpecialPwd#2025"
host="localhost"
port="3306"
#需备份的数据库，数组
db=(ekp kk aip)
#备份时加锁方式，
#MyISAM为锁表--lock-all-tables，
#InnoDB为锁行--single-transaction
lock="--single-transaction"
mysql_path="/usr/local/mysql"
backup_path="/backup/mysql_bak"
date=$(date +%Y-%m-%d_%H-%M-%S)
day=15
backup_log="${backup_path}/backup.log"

#建立备份目录
if [ ! -e $backup_path ];then
    mkdir -p $backup_path
fi

#删除以前备份
find $backup_path -type f -mtime +$day -exec rm -rf {} \; > /dev/null 2>&1

echo "开始备份数据库：${db[*]}"

#备份并压缩
backup_sql(){
    dbname=$1
    backup_name="${dbname}_${date}.sql"
    #-R备份存储过程，函数，触发器
    mysqldump -S /data/mysql/mysql.sock -P $port -u $user -p$password $lock --default-character-set=utf8 --opt --flush-logs --set-gtid-purged=OFF --events  --triggers  --routines    --databases  $dbname > $backup_path/$backup_name    
    if [[ $? == 0 ]];then
        cd $backup_path
        tar zcpvf $backup_name.tar.gz $backup_name
        size=$(du $backup_name.tar.gz -sh | awk '{print $1}')
        rm -rf $backup_name
        echo "$date 备份 $dbname($size) 成功 "
    else
        cd $backup_path
        rm -rf $backup_name
        echo "$date 备份 $dbname 失败 "
    fi
}

#循环备份
length=${#db[@]}
for (( i = 0; i < $length; i++ )); do
        backup_sql ${db[$i]} >> $backup_log 2>&1
done
```

```bash
#!/bin/sh
# File: /ap/bin/linux_mysql.sh

# Database info
DB_HOST="localhost"
DB_PORT="3306"
DB_NAME="ekptest"
DB_USER="root"
DB_PASS="Super@99"

# 备份保留天数
BACK_TIME="7"

# 备份路径
BAK_DIR="/data/backup"

# 日志文件
Logfile="backup_ekp_db.log"

# ===================== 核心设置：你自由控制CPU =====================
# 压缩使用 CPU 核心数（比如 4、6、8、12，根据你服务器配置填）
COMPRESS_THREADS=6
# =====================================================================

DATE=`date +%F`
time=`date +%Y-%m-%d" "%H:%M:%S`

if [ $(echo $0 | grep '^/') ]; then
    path=$(dirname $0)
else
    path=$(pwd)/$(dirname $0)
fi

cd $path

if [ ! -d $BAK_DIR ]; then
  mkdir -p $BAK_DIR
fi

# ===================== 备份命令（多核压缩） =====================
mysqldump -S /data/mysql/mysql.sock -P $DB_PORT -p$DB_PASS --opt -u$DB_USER $DB_NAME \
| pigz -$COMPRESS_THREADS > $BAK_DIR/ekp_db_$DATE.gz

if [ $? -eq 0 ]
then
    echo $time "Backup database of EKP success~! " ekp_db_$DATE.gz >> $Logfile

    echo $time "backup database of EKP failed~!">>$Logfile
fi

find $BAK_DIR -name "*.gz" -mtime +$BACK_TIME -type f |xargs rm -f

if [ $? -eq 0 ]
then
    echo $time "Delete history backup file of" $BACK_TIME " days ago success~!">>$Logfile
else
    echo $time "delete files failed">>$Logfile
fi
```
